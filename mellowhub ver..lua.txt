local Players = game:GetService("Players")
local UserInputService = game:GetService("UserInputService")

local player = Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()
local humanoid = character:WaitForChild("Humanoid")


local ScreenGui = Instance.new("ScreenGui")
local Frame = Instance.new("Frame")
local TeleportButton = Instance.new("TextButton")
local InvisibleButton = Instance.new("TextButton")
local FlyButton = Instance.new("TextButton")
local WalkSpeedSlider = Instance.new("TextBox")
local JumpPowerSlider = Instance.new("TextBox")

ScreenGui.Parent = player:WaitForChild("PlayerGui")
Frame.Parent = ScreenGui

Frame.Size = UDim2.new(0, 200, 0, 300)
Frame.Position = UDim2.new(0, 10, 0, 10)
Frame.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
Frame.BorderSizePixel = 2


local function setupButton(button, text, position)
    button.Size = UDim2.new(0, 180, 0, 30)
    button.Position = position
    button.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.Text = text
    button.Parent = Frame
end

setupButton(TeleportButton, "Teleport", UDim2.new(0, 10, 0, 10))
setupButton(InvisibleButton, "Invisible", UDim2.new(0, 10, 0, 50))
setupButton(FlyButton, "Fly", UDim2.new(0, 10, 0, 90))


TeleportButton.MouseButton1Click:Connect(function()
    local destination = Vector3.new(0, 50, 0) 
    character:MoveTo(destination)
end)

InvisibleButton.MouseButton1Click:Connect(function()
    for _, part in ipairs(character:GetDescendants()) do
        if part:IsA("BasePart") and part.Name ~= "HumanoidRootPart" then
            part.Transparency = 1
            part.CanCollide = false
        end
    end
end)


local flying = false
FlyButton.MouseButton1Click:Connect(function()
    flying = not flying
    if flying then
        humanoid.PlatformStand = true
        local flyConnection
        flyConnection = UserInputService.InputBegan:Connect(function(input)
            if input.KeyCode == Enum.KeyCode.W then
                character:TranslateBy(Vector3.new(0, 5, 0))
            elseif input.KeyCode == Enum.KeyCode.S then
                character:TranslateBy(Vector3.new(0, -5, 0))
            end
        end)
    else
        humanoid.PlatformStand = false
    end
end)


local function createTextBox(textbox, text, position)
    textbox.Size = UDim2.new(0, 180, 0, 30)
    textbox.Position = position
    textbox.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    textbox.TextColor3 = Color3.fromRGB(255, 255, 255)
    textbox.Text = text
    textbox.Parent = Frame
end

createTextBox(WalkSpeedSlider, "WalkSpeed: 16", UDim2.new(0, 10, 0, 130))
createTextBox(JumpPowerSlider, "JumpPower: 50", UDim2.new(0, 10, 0, 170))

WalkSpeedSlider.FocusLost:Connect(function()
    local speed = tonumber(WalkSpeedSlider.Text:match("%d+"))
    if speed then
        humanoid.WalkSpeed = speed
    end
end)

JumpPowerSlider.FocusLost:Connect(function()
    local power = tonumber(JumpPowerSlider.Text:match("%d+"))
    if power then
        humanoid.JumpPower = power
    end
end)
